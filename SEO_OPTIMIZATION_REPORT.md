## 2. Technical SEO Implementation (Next.js 14)

### A. Sitemap & Robots.txt
Next.js App Router has built-in support for these.

**Action**: Create `src/app/sitemap.ts` and `src/app/robots.ts`.

```typescript
// src/app/sitemap.ts
import { MetadataRoute } from 'next';

export default function sitemap(): MetadataRoute.Sitemap {
  const baseUrl = 'https://www.placenxt.com';
  
  return [
    { url: baseUrl, lastModified: new Date(), changeFrequency: 'daily', priority: 1 },
    { url: `${baseUrl}/pricing`, lastModified: new Date(), changeFrequency: 'weekly', priority: 0.8 },
    { url: `${baseUrl}/login`, lastModified: new Date(), changeFrequency: 'monthly', priority: 0.5 },
    // Add blog posts or dynamic routes here
  ]
}
```

```typescript
// src/app/robots.ts
import { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: ['/dashboard/', '/admin/'], // Hide private pages
    },
    sitemap: 'https://placenxt.com/sitemap.xml',
  }
}
```
